#include<bits/stdc++.h>
using namespace std ;
const int N = 110 , M = 10010 , INF = 0x3f3f3f3f ;
typedef long long LL ;
int n, m;
int g[N][N];
int d[N][N];  // æœ€çŸ­è·ç¦»

int path[N], cnt;  // å½“å‰æœ€å°ç¯çš„æ–¹æ¡ˆ

void get_path(int i, int j) {
    if (pos[i][j] == 0) return;
    int k = pos[i][j];
    get_path(i, k);
    path[cnt ++] = k;
    get_path(k, j);
}

int main() {
    cin >> n >> m;

    memset(g, 0x3f, sizeof g);
    for (int i = 1; i <= n; i ++) g[i][i] = 0;

    while (m --) {
        int a, b, c;
        cin >> a >> b >> c;
        g[a][b] = g[b][a] = min(g[a][b], c);
    }

    int res = INF;
    memcpy(d, g, sizeof g);
    for (int k = 1; k <= n; ++ k) {
        // æšä¸¾ä»¥ k ä¸ºæœ€å¤§ç¼–å·çš„ç¯
        // ç”±äºæ˜¯æ— å‘å›¾ï¼Œi j é¡ºåºå¯äº’æ¢ï¼Œæ‰€ä»¥ä¸ç”¨éƒ½æšä¸¾
        for (int i = 1; i < k; ++ i)
            for (int j = i + 1; j < k; ++ j)
                if ((LL)d[i][j] + g[i][k] + g[k][j] < res) {
                    res = d[i][j] + g[i][k] + g[k][j];
                    cnt = 0;
                    path[cnt ++] = k;
                    path[cnt ++] = i;
                    get_path(i, j);
                    path[cnt ++] = j;
                }
        // floyd
        for (int i = 1; i <= n; ++ i)
            for (int j = 1; j <= n; ++ j)
                if (d[i][j] > d[i][k] + d[k][j]) {
                    d[i][j] = d[i][k] + d[k][j];
                    pos[i][j] = k;
                }
    }

    if (res == INF) puts("No solution.");
    else {
        for (int i = 0; i < cnt; ++ i)
            cout << path[i] << ' ';
        cout << endl;
    }
}
